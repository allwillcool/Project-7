#include <stdio.h>
#include <math.h>
#include <windows.h>

int main() {
     SetConsoleCP(1251);
  SetConsoleOutputCP(1251);
    double x1, y1, r1, x2, y2, r2;
    int intersectionPoints = 0;

    printf("Введіть координати та радіус першого кола, наприклад: (x1, y1, r1): ");
    scanf("%lf %lf %lf", &x1, &y1, &r1);

    printf("Введіть координати та радіус другого кола, наприклад: (x2, y2, r2): ");
    scanf("%lf %lf %lf", &x2, &y2, &r2);

    // Обчислюємо відстань між центрами колів
    double distance = sqrt(pow(x2 - x1, 2) + pow(y2 - y1, 2));

    // Перевіряємо умови перетину колів
    if (distance == 0 && r1 == r2) {
        // Кола співпадають, маємо безкінечну кількість точок перетину
        intersectionPoints = -1;
    } else if (distance <= r1 + r2 && distance >= fabs(r1 - r2)) {
        // Кола перетинаються в декількох точках
        intersectionPoints = 2;
    } else {
        // Кола не перетинаються
        intersectionPoints = 0;
    }

    printf("Кількість точок перетину: %d\n", intersectionPoints);

    return 0;
}
